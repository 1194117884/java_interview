# ✅图片优化的常见手段

# 典型回答

图片通常是网页中体积最大的资源，优化图片可以显著提升页面加载速度。常见的图片优化手段分为以下几个方面：

**1. 选择正确的图片格式：**
| 格式 | 适用场景 | 特点 |
|------|---------|------|
| JPEG | 照片、复杂渐变图 | 有损压缩，不支持透明 |
| PNG | 图标、截图、需透明的图形 | 无损压缩，支持透明，文件较大 |
| WebP | 现代浏览器（96%+支持率） | 有损/无损皆可，支持透明，比JPEG小25-35% |
| AVIF | 最新格式（90%+支持率） | 更优的压缩率，比WebP小20-30% |
| SVG | 图标、Logo、矢量图形 | 可缩放，文件极小（适用于简单图形） |

**2. 压缩和编码优化：** 图片压缩分为有损（牺牲画质换取更小体积）和无损（保持原始画质）两种方式。使用工具或 CDN 服务实现自动化压缩。

**3. 响应式图片：** 根据设备屏幕大小和分辨率加载不同尺寸的图片，避免手机加载桌面大图。

**4. 懒加载技术：** 图片进入视口时才加载，减少首屏传输字节数。

**5. CDN 加速：** 通过 CDN 的边缘节点就近提供图片，减少网络延迟。

### 响应式图片实现

```html
<!-- 方案1: srcset + sizes 属性 -->
<!-- 根据视口宽度选择不同尺寸的图片 -->
<img 
  src="hero-800.jpg" 
  srcset="
    hero-400.jpg 400w,
    hero-800.jpg 800w, 
    hero-1200.jpg 1200w,
    hero-1600.jpg 1600w"
  sizes="
    (max-width: 600px) 100vw,
    (max-width: 1024px) 80vw,
    1200px"
  alt="横幅"
  width="1200" 
  height="600"
  decoding="async"
>

<!-- 方案2: <picture> 元素（支持不同格式和裁切） -->
<picture>
  <!-- 优先使用 AVIF 格式 -->
  <source 
    srcset="hero.avif" 
    type="image/avif"
    media="(min-width: 768px)">
  
  <!-- 备选 WebP 格式 -->
  <source 
    srcset="hero.webp" 
    type="image/webp"
    media="(min-width: 768px)">
  
  <!-- 移动端使用不同裁切版本 -->
  <source 
    srcset="hero-mobile.webp" 
    type="image/webp"
    media="(max-width: 767px)">
  
  <!-- 最坏情况降级 -->
  <img src="hero.jpg" alt="横幅" width="1200" height="600">
</picture>

<!-- 方案3: 根据屏幕密度选择 -->
<img 
  src="icon.png"
  srcset="icon@2x.png 2x, icon@3x.png 3x"
  alt="图标"
>
```

### 图片懒加载

```html
<!-- 方案1: 原生懒加载（推荐） -->
<img src="hero.jpg" loading="lazy" alt="横幅">

<!-- 方案2: 结合 IntersectionObserver -->
<img 
  class="lazy" 
  data-src="real-image.jpg" 
  src="placeholder.svg"  <!-- 占位图 -->
  alt="需要加载的图片"
>

<script>
  // IntersectionObserver 实现懒加载
const lazyImages = document.querySelectorAll('.lazy');

const imageObserver = new IntersectionObserver((entries, observer) => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      const img = entry.target;
      
      // 替换真实图片
      img.src = img.dataset.src;
      img.srcset = img.dataset.srcset || '';
      
      // 移除 class 触发渐入动画
      img.classList.add('loaded');
      
      // 停止观察
      observer.unobserve(img);
    }
  });
}, {
  rootMargin: '200px 0px',  // 提前 200px 加载（预加载）
  threshold: 0.01,
});

lazyImages.forEach(img => imageObserver.observe(img));
</script>

<style>
/* 懒加载渐入动画 */
.lazy {
  opacity: 0;
  transition: opacity 0.3s ease-in;
}
.lazy.loaded {
  opacity: 1;
}
</style>
```

```javascript
// 使用 CDN 图片处理服务（以 Cloudinary 为例）
// https://res.cloudinary.com/demo/image/upload/v1/sample.jpg

// 动态参数处理图片
const baseUrl = 'https://res.cloudinary.com/demo/image/upload';

function buildImageUrl(publicId, options = {}) {
  const params = new URLSearchParams();
  
  // 宽度和高度
  if (options.width) params.append('w', options.width);
  if (options.height) params.append('h', options.height);
  
  // 裁切模式
  if (options.fit) params.append('c', options.fit); // fill, fit, crop, pad
  
  // 质量
  params.append('q', options.quality || 'auto');
  
  // 格式（自动选择最佳格式）
  params.append('f', 'auto');
  
  // 锐化
  if (options.sharpen) params.append('e_sharpen', options.sharpen);
  
  return `${baseUrl}/${publicId}?${params.toString()}`;
}

// 使用示例
const mobileHero = buildImageUrl('hero.jpg', {
  width: 400,
  quality: 80,
  fit: 'crop',
});

const desktopHero = buildImageUrl('hero.jpg', {
  width: 1200,
  quality: 75,
  fit: 'fill',
});
```

### 图片的缓存策略

```nginx
# Nginx 图片缓存配置
location ~* \.(jpg|jpeg|png|gif|ico|webp|avif|svg)$ {
    # 内容变更时通过修改 URL 来刷新缓存
    expires 30d;
    add_header Cache-Control "public, immutable";
    
    # 开启 gzip（对于 SVG 很有用）
    gzip on;
    gzip_types image/svg+xml;
    
    # 关闭访问日志（图片请求量通常很大）
    access_log off;
    
    # 开启 open_file_cache
    open_file_cache max=1000 inactive=20s;
}
```

### 图片性能检测

```javascript
// 检测页面中图片的优化情况
function auditImages() {
  const images = document.querySelectorAll('img');
  const issues = [];
  
  images.forEach(img => {
    const issue = {};
    
    // 是否缺少宽高（影响 CLS）
    if (!img.hasAttribute('width') || !img.hasAttribute('height')) {
      issue.missingDimensions = true;
    }
    
    // 是否使用了现代格式
    const src = img.src || '';
    if (src.match(/\.(jpg|jpeg|png)$/i)) {
      issue.oldFormat = true;
    }
    
    // 是否使用了懒加载（首屏图片不应懒加载）
    if (img.hasAttribute('loading') && img.getAttribute('loading') === 'lazy') {
      const rect = img.getBoundingClientRect();
      if (rect.top < window.innerHeight) {
        issue.lazyOnAboveFold = true;
      }
    }
    
    // 是否使用了 srcset
    if (!img.hasAttribute('srcset')) {
      issue.missingSrcset = true;
    }
    
    // 文件大小估算（通过 Resource Timing API）
    const resourceEntry = performance.getEntriesByName(img.src)[0];
    if (resourceEntry) {
      const sizeKB = (resourceEntry.transferSize || resourceEntry.encodedBodySize) / 1024;
      if (sizeKB > 100) {
        issue.largeFile = `${sizeKB.toFixed(1)}KB`;
      }
    }
    
    if (Object.keys(issue).length > 0) {
      issues.push({ src: img.src, ...issue });
    }
  });
  
  console.table(issues);
}

// 观察 LCP 元素是否为图片
new PerformanceObserver((list) => {
  const entries = list.getEntries();
  const lcp = entries[entries.length - 1];
  
  if (lcp.element && lcp.element.tagName === 'IMG') {
    console.log('LCP 图片:', lcp.element.src);
    console.log('LCP 图片大小:', lcp.element.naturalWidth, 'x', lcp.element.naturalHeight);
  }
}).observe({ type: 'largest-contentful-paint', buffered: true });
```

### 图片优化清单

| 优化项 | 优先级 | 预期效果 |
|--------|--------|---------|
| 使用 WebP/AVIF 格式 | 高 | 减少体积 25-35% |
| 响应式图片（srcset） | 高 | 移动端减少 50-80% 传输量 |
| 有损压缩（quality 75-85） | 高 | 减少体积 40-60% |
| 懒加载（loading="lazy"） | 中 | 首屏减少 50%+ 图片请求 |
| 使用 CDN | 中 | 减少延迟 50-80% |
| 渐进式 JPEG | 中 | 改善用户体验 |
| 占位符（LQIP/SQIP） | 低 | 减少布局偏移 |
| 图片 Sprite（雪碧图） | 低（HTTP/2 后不再推荐） | - |
| inline base64 小图 | 低（有缓存和时间开销） | - |
