# ✅闭包是什么？有什么应用场景和缺点？

# 典型回答

**闭包（Closure）** 是指一个函数"记忆"并访问其词法作用域的能力，即使该函数在其词法作用域之外执行。闭包是**函数**和**创建该函数的环境**的组合。

```javascript
function outer() {
  const message = 'Hello';
  function inner() {
    console.log(message);  // inner引用了outer的变量
  }
  return inner;
}

const closure = outer();
closure(); // 'Hello' — 闭包维持了对message的引用
```

**闭包的核心特性：**
- 内部函数可以访问外部函数的变量
- 外部函数的变量不会被垃圾回收（只要内部函数存在）
- 每次调用外部函数都会创建新的闭包

**常见应用场景：**

| 场景 | 说明 |
|------|------|
| 数据私有化 | 模拟私有变量和方法 |
| 柯里化 | 将多参数函数转换为单参数函数链 |
| 函数工厂 | 创建带有预设参数的函数 |
| 事件处理 | 绑定点击事件时保持状态 |
| 防抖节流 | 保存定时器状态 |
| 模块模式 | 实现模块化封裝 |
| 高阶函数 | 函数作为返回值 |

**闭包的缺点：**
- 内存占用：闭包阻止垃圾回收，可能导致内存泄漏
- 性能开销：多层闭包增加了作用域链查找的层次
- 调试困难：闭包中的变量在调试时不易追踪

# 扩展知识

## 闭包的工作原理

```javascript
// 每个函数在执行时都会创建[[Scopes]]属性
function outer() {
  const x = 10;
  const y = 20;
  function inner() {
    return x + y;  // inner通过作用域链访问x和y
  }
  return inner;
}

const fn = outer();
console.log(fn()); // 30

// 可视化表示
// fn.[[Scopes]] = [outer函数的变量对象, 全局变量对象]
// 即使outer执行完毕，outer的作用域对象仍然被inner引用，不会被GC回收
```

## 闭包与循环

```javascript
// 经典问题：闭包捕获的是变量的引用而非值
for (var i = 0; i < 5; i++) {
  setTimeout(function() {
    console.log(i); // 5, 5, 5, 5, 5
  }, 100);
}

// 解决方案1：IIFE创建独立作用域
for (var i = 0; i < 5; i++) {
  (function(j) {
    setTimeout(function() {
      console.log(j); // 0, 1, 2, 3, 4
    }, 100);
  })(i);
}

// 解决方案2：使用let（块级作用域）
for (let i = 0; i < 5; i++) {
  setTimeout(function() {
    console.log(i); // 0, 1, 2, 3, 4
  }, 100);
}

// 解决方案3：使用闭包工厂函数
function createCallback(i) {
  return function() {
    console.log(i);
  };
}
for (var i = 0; i < 5; i++) {
  setTimeout(createCallback(i), 100);
}
```

## 模拟私有变量

```javascript
// 使用闭包实现私有变量
function createCounter() {
  let count = 0;  // 私有变量

  return {
    increment() {
      count++;
      return count;
    },
    decrement() {
      count--;
      return count;
    },
    getCount() {
      return count;
    }
  };
}

const counter = createCounter();
console.log(counter.increment()); // 1
console.log(counter.increment()); // 2
console.log(counter.getCount());  // 2
// console.log(counter.count);    // undefined — 外部无法直接访问

// 使用Symbol实现类似效果
const _count = Symbol('count');
class Counter {
  constructor() {
    this[_count] = 0;
  }
  increment() { return ++this[_count]; }
}
// 但Object.getOwnPropertySymbols()仍然可以获取
```

## 模块模式

```javascript
// 闭包实现模块化
const myModule = (function() {
  // 私有变量
  const privateData = [];
  let id = 0;

  // 私有函数
  function validate(item) {
    return item !== null && item !== undefined;
  }

  // 公共API
  return {
    add(item) {
      if (validate(item)) {
        privateData.push({ id: id++, value: item });
        return true;
      }
      return false;
    },
    getAll() {
      return [...privateData];
    },
    getById(itemId) {
      return privateData.find(item => item.id === itemId);
    },
    remove(itemId) {
      const index = privateData.findIndex(item => item.id === itemId);
      if (index !== -1) {
        privateData.splice(index, 1);
        return true;
      }
      return false;
    },
    get count() {
      return privateData.length;
    }
  };
})();

myModule.add('item1');
console.log(myModule.count); // 1
console.log(myModule.getAll()); // [{ id: 0, value: 'item1' }]
```

## 函数柯里化

```javascript
// 闭包实现柯里化
function curry(fn) {
  return function curried(...args) {
    if (args.length >= fn.length) {
      return fn.apply(this, args);
    }
    return function(...args2) {
      return curried.apply(this, args.concat(args2));
    };
  };
}

// 普通函数
function sum(a, b, c) {
  return a + b + c;
}

const curriedSum = curry(sum);
console.log(curriedSum(1)(2)(3));   // 6
console.log(curriedSum(1, 2)(3));   // 6
console.log(curriedSum(1)(2, 3));   // 6

// 实际应用：日志函数
function log(level) {
  return function(message) {
    console.log(`[${level.toUpperCase()}] ${message}`);
  };
}

const info = log('info');
const warn = log('warn');
const error = log('error');

info('系统启动');   // [INFO] 系统启动
error('连接失败');  // [ERROR] 连接失败
```

## 防抖和节流

```javascript
// 防抖：多次触发只执行最后一次
function debounce(fn, delay) {
  let timer = null;  // 闭包保存timer
  return function(...args) {
    const context = this;
    clearTimeout(timer);
    timer = setTimeout(() => {
      fn.apply(context, args);
    }, delay);
  };
}

// 节流：限制执行频率
function throttle(fn, limit) {
  let inThrottle = false;  // 闭包保存状态
  return function(...args) {
    const context = this;
    if (!inThrottle) {
      fn.apply(context, args);
      inThrottle = true;
      setTimeout(() => { inThrottle = false; }, limit);
    }
  };
}

// 使用
const handleScroll = throttle(() => {
  console.log('滚动事件处理');
}, 200);
window.addEventListener('scroll', handleScroll);
```

## 闭包的内存管理

```javascript
// 不必要的闭包导致内存泄漏
function createLargeArray() {
  const largeArray = new Array(1000000).fill('data');

  return function() {
    console.log(largeArray.length);
    // largeArray无法被GC回收，因为闭包一直引用它
  };
}

const fn = createLargeArray();
// 即使fn只用一次，largeArray依然占用内存

// 手动解除引用
fn = null;  // 闭包被释放，largeArray可被GC回收

// 优化：只保留需要的数据
function createEfficient() {
  const largeArray = new Array(1000000).fill('data');
  const length = largeArray.length;  // 只保留需要的值

  return function() {
    console.log(length);
  };
}
```

## 闭包与作用域链的交互

```javascript
// 多层嵌套闭包共享同一作用域的问题
function createFunctions() {
  const functions = [];
  for (var i = 0; i < 5; i++) {
    functions.push(function() {
      console.log(i);
    });
  }
  return functions;
}
const fns = createFunctions();
fns[0](); // 5 — 所有函数共享同一个i
fns[1](); // 5

// 用IIFE创建独立作用域
function createFunctionsFixed() {
  const functions = [];
  for (var i = 0; i < 5; i++) {
    (function(j) {
      functions.push(function() {
        console.log(j);
      });
    })(i);
  }
  return functions;
}
```

## React Hooks中的闭包陷阱

```javascript
// React函数组件中常见的闭包问题
function Counter() {
  const [count, setCount] = useState(0);

  useEffect(() => {
    const timer = setInterval(() => {
      console.log(count); // 总是0，因为闭包捕获的是初始值
      setCount(count + 1); // count始终是0，所以始终设为1
    }, 1000);

    return () => clearInterval(timer);
  }, []);

  // 修复：使用函数式更新
  // setCount(prev => prev + 1);
  // 或将count加入依赖数组
}
```
