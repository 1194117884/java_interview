# ✅作用域链和变量提升机制

# 典型回答

## 作用域链（Scope Chain）

**作用域链**是JavaScript在查找变量时遵循的层级查找机制。当代码访问一个变量时，JavaScript引擎会从当前作用域开始查找，如果找不到则逐级向上级作用域查找，直到全局作用域。如果全局作用域也找不到，在严格模式下抛出ReferenceError，非严格模式下返回undefined。

```javascript
const globalVar = 'global';

function outer() {
  const outerVar = 'outer';

  function inner() {
    const innerVar = 'inner';
    console.log(innerVar);  // 'inner' — 当前作用域
    console.log(outerVar);  // 'outer' — 上级作用域
    console.log(globalVar); // 'global' — 全局作用域
  }

  inner();
}

outer();
// 作用域链查找顺序: inner → outer → global
```

## 变量提升（Hoisting）

**变量提升**是JavaScript在代码执行前将变量和函数声明"移动"到作用域顶部的行为。这实际上是JavaScript引擎在编译阶段处理声明的方式。

```javascript
console.log(a); // undefined（不会报错）
var a = 1;

// 上述代码等价于：
var a;
console.log(a); // undefined
a = 1;
```

# 扩展知识

## 词法作用域（静态作用域）

JavaScript采用**词法作用域**（也叫静态作用域），即作用域在代码编写时就已经确定，而不是在运行时：

```javascript
const value = 'global';

function foo() {
  console.log(value);
}

function bar() {
  const value = 'bar';
  foo(); // 输出 'global'，不是 'bar'
}

bar();
// foo的作用域在定义时确定，与调用位置无关
```

## 作用域的类型

```javascript
// 1. 全局作用域
const globalVar = 'global';  // 任何地方都可访问

// 2. 函数作用域
function fn() {
  const fnVar = 'function scope';
  // fnVar只能在函数内部访问
}

// 3. 块级作用域（ES6，let/const）
{
  let blockVar = 'block';
  const blockConst = 'block const';
  var notBlockVar = 'not block';  // var没有块级作用域
}
console.log(notBlockVar);  // 'not block'
// console.log(blockVar);  // ReferenceError

// 4. 模块作用域（ES Module）
// 每个模块都有自己的作用域，不会污染全局
```

## 变量提升的详细规则

```javascript
// var的提升
console.log(x); // undefined（声明提升，但未赋值）
var x = 10;

// 等价于：
var x;
console.log(x); // undefined
x = 10;

// 函数声明的提升（整体提升）
sayHi(); // 'Hi!' — 函数声明会整体提升

function sayHi() {
  console.log('Hi!');
}

// 函数表达式的提升（只有变量声明被提升）
// sayHello(); // TypeError: sayHello is not a function

var sayHello = function() {
  console.log('Hello!');
};

// 等价于：
var sayHello;
// sayHello(); // 此时sayHello是undefined，不是函数
sayHello = function() {};
```

## let/const没有"真正"的提升

```javascript
// 表面上看起来没有提升，实际上有提升但存在暂时性死区
console.log(a); // ReferenceError: Cannot access 'a' before initialization
let a = 1;

// 证明let有提升
let x = 'outer';
{
  console.log(x); // ReferenceError（而不是输出'outer'）
  let x = 'inner';
}
// 如果let没有提升，应该输出'outer'，但报错说明x被提升了但处于TDZ
```

## 执行上下文与作用域链的创建

```javascript
// 每个函数调用都会创建新的执行上下文
// 执行上下文包含：变量环境、词法环境、this绑定、外部环境引用

const a = 'global';

function outer(p) {
  const b = 'outer';

  function inner() {
    const c = 'inner';
    console.log(a, b, c, p); // 可以访问所有外层变量+参数
  }

  return inner;
}

const innerFn = outer('param');
innerFn();

// 作用域链：[inner的作用域, outer的作用域, 全局作用域]
// 闭包：innerFn维持了对outer作用域的引用
```

## 作用域链的优化

```javascript
// 避免过深的作用域链嵌套
// 不佳实践
function level1() {
  const a = 1;
  function level2() {
    const b = 2;
    function level3() {
      const c = 3;
      // 访问a需要跨越3层作用域链
      console.log(a + b + c);
    }
    level3();
  }
  level2();
}

// 更好的方式：显式传递参数
function better(a, b, c) {
  console.log(a + b + c);
}

// 使用with（不推荐）会改变作用域链
const obj = { x: 10, y: 20 };
with (obj) {
  console.log(x + y); // 30 — 动态添加obj到作用域链前端
  // 严格模式下禁用
}
```

## 作用域链图解

```
全局作用域
  │
  ├── 变量: globalVar
  │
  └── outer函数作用域
        │
        ├── 变量: outerVar, inner函数引用
        │
        └── inner函数作用域
              │
              └── 变量: innerVar
```

## 常见面试题

```javascript
// 题1：经典循环
for (var i = 0; i < 5; i++) {
  setTimeout(() => console.log(i), 0); // 5, 5, 5, 5, 5
}

for (let i = 0; i < 5; i++) {
  setTimeout(() => console.log(i), 0); // 0, 1, 2, 3, 4
}

// 题2：变量提升顺序
console.log(typeof foo);   // 'function'
console.log(typeof bar);   // 'undefined'

var foo = 'string';
var bar = function() {};

function foo() { return 'function'; }

// 等价于：
function foo() { return 'function'; }
var bar;
// console.log(typeof foo); // 'function'（函数声明优先级高于变量声明）
// console.log(typeof bar); // 'undefined'
bar = function() {};
foo = 'string';

// 题3：作用域链与参数
var x = 10;
function test(x) {
  console.log(x); // 5（参数覆盖了全局变量）
  var x = 20;
  console.log(x); // 20
}
test(5);

// 题4：嵌套作用域
var a = 1;
function fn1() {
  var a = 2;
  function fn2() {
    var a = 3;
    console.log(a); // 3
  }
  fn2();
  console.log(a); // 2
}
fn1();
console.log(a); // 1
```
