# ✅什么是暂时性死区？

# 典型回答

**暂时性死区（Temporal Dead Zone, TDZ）** 是ES6中引入的概念，指的是从**作用域顶部**到**变量声明语句**之间的区域。在这个区域内，变量虽然已经被声明（作用域知道这个变量存在），但还不能被访问，访问会抛出 `ReferenceError`。

```javascript
{
  // TDZ 开始（对于变量 val）
  console.log(typeof val);  // ReferenceError（不是undefined！）
  let val = 1;              // TDZ 结束
  console.log(val);         // 1 — 可以访问
}
```

**核心要点：**
- let和const存在暂时性死区，var不存在
- 在TDZ中访问变量会抛出ReferenceError
- typeof操作符在TDZ中也不再安全
- TDZ的目的是防止变量在声明前被意外使用
- 在声明语句之前，变量处于"未初始化"状态

# 扩展知识

## 暂时性死区的产生原因

JavaScript引擎在执行代码前会进行编译（预处理）。对于let和const声明的变量，引擎知道它们存在于当前作用域，但不会像var那样初始化为undefined：

```javascript
// var的处理方式：声明提升 + 初始化为undefined
console.log(a); // undefined
var a = 1;

// let的处理方式：声明提升但未初始化
console.log(b); // ReferenceError: Cannot access 'b' before initialization
let b = 2;
```

let/const的声明在编译阶段被识别，但在执行到声明语句前，变量处于"未初始化"状态（uninitialized），这个状态形成了TDZ。

## 不同变量类型的TDZ对比

```javascript
// var：没有TDZ，初始化为undefined
console.log(varVal); // undefined（编译阶段提升并初始化）
var varVal = 'var';

// let：有TDZ
console.log(letVal); // ReferenceError（提升但未初始化）
let letVal = 'let';

// const：有TDZ，且必须在声明时赋值
console.log(constVal); // ReferenceError（提升但未初始化）
const constVal = 'const';

// class：也有TDZ
const p = new MyClass(); // ReferenceError
class MyClass {}

// 函数声明：没有TDZ，整个提升
foo(); // '函数声明' — 完全提升
function foo() { console.log('函数声明'); }
```

## 常见TDZ面试题

```javascript
// 题1：经典TDZ
let x = 'outer';
{
  console.log(x); // ReferenceError（不是'outer'）
  let x = 'inner';
}
// let x 在块级作用域内提升了，访问的是块内的x，不是外层的x

// 题2：typeof与TDZ
typeof notDeclared;   // 'undefined' — 未声明变量是安全的
typeof y;             // ReferenceError — TDZ中的变量
let y;

// 题3：默认参数中的TDZ
function fn(a = b, b = 1) {
  // 参数a的默认值b处于TDZ
}
// fn(); // ReferenceError: Cannot access 'b' before initialization

// 正确的方式：使用已经声明的参数
function fn(a = 1, b = a) {
  console.log(a, b);
}
fn(); // 1 1

// 题4：函数参数默认值中的内部TDZ
let x = 1;
function fn(x = x) {  // 参数x的默认值引用了x本身
  // 参数x的TDZ：在参数初始化完成前访问自身
}
// fn(); // ReferenceError: Cannot access 'x' before initialization
```

## TDZ在条件语句中的表现

```javascript
// TDZ按照实际执行流程确定
let condition = true;

if (condition) {
  console.log(value); // ReferenceError: Cannot access 'value' before initialization
  let value = 42;
}

// 如果条件为false，永远不进入块，就不会有TDZ
if (false) {
  // 这块代码不执行，所以不会触发TDZ
  let value = 42;
  console.log(value);
}

// TDZ受执行顺序影响
function test(condition) {
  if (condition) {
    console.log(msg); // TDZ
    return;
  }
  let msg = 'hello';
}
// test(true); // ReferenceError
```

## TDZ与try/catch

```javascript
// 即使有错误处理也无法绕过TDZ
try {
  console.log(val); // ReferenceError
  let val = 1;
} catch (e) {
  console.log('捕获:', e.message); // 这行会被执行
}

// 与eval配合
eval('console.log(x); let x = 1;'); // ReferenceError（eval中的TDZ）

// 与with配合（严格模式下禁用）
```

## TDZ与继承

```javascript
class Parent {
  constructor() {
    this.name = 'parent';
  }
}

class Child extends Parent {
  constructor() {
    // console.log(this); // ReferenceError: must call super before accessing this
    super();
    // this现在可以访问了
    console.log(this.name); // 'parent'
  }
}

// 在子类构造函数中，在调用super()之前，this处于TDZ状态
```

## TDZ的实际意义

```javascript
// 1. 防止使用未声明的变量（编程错误）
function bad() {
  console.log(undeclared); // 假设想用某个值但还没声明
  let undeclared = 1;
}

// 2. 使代码更符合直觉（先声明再使用）
function good() {
  let value = 1;
  console.log(value); // 1 — 符合"先声明后使用"的直觉
}

// 3. 更容易发现变量提升导致的bug
// 没有TDZ时：
console.log(name); // undefined（你可能以为会报错）
var name = 'Alice'; // 但实际上这是一个无意义的变量访问

// TDZ确保了代码逻辑更清晰
```

## var与TDZ的对比

```javascript
// var没有TDZ，但会带来一些困扰
var x = 1;

function test() {
  // x提升了，但没有TDZ
  console.log(x); // undefined（不是1也不是ReferenceError）
  var x = 2;
}
test();

// 对比let：
function test2() {
  console.log(x); // ReferenceError — 明确告诉你变量声明前不能使用
  let x = 2;
}
test2();

// TDZ让bug更容易被发现
```

## 开发建议

- 始终在作用域顶部声明变量，避免TDZ触发
- 使用const作为默认选择，let在需要重新赋值时使用
- 避免在声明前访问变量
- 函数参数默认值中注意参数的依赖顺序
- class声明也存在TDZ，注意引用顺序
