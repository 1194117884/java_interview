# ✅并发控制：如何限制异步操作的并发数？

# 典型回答

限制异步操作的并发数是通过**控制同时执行的任务数量**，避免过多并发请求导致资源耗尽（如服务器压力、浏览器连接数限制等）。

**核心实现思路：**

1. **维护一个"进行中"的任务池**，限制池中任务数量不超过最大并发数
2. **当一个任务完成时，从任务队列中取出下一个任务执行**
3. **确保所有任务最终都会被执行**

**基本实现：**

```javascript
async function concurrentControl(tasks, limit) {
  const results = [];
  const executing = [];

  for (const [index, task] of tasks.entries()) {
    // 创建任务Promise，并记录其索引
    const p = Promise.resolve().then(() => task());
    results[index] = p;

    // 任务完成后从执行池中移除
    const e = p.then(() => executing.splice(executing.indexOf(e), 1));
    executing.push(e);

    // 如果正在执行的任务达到限制，等待任意一个完成
    if (executing.length >= limit) {
      await Promise.race(executing);
    }
  }

  // 等待所有任务完成并返回结果
  return Promise.all(results);
}
```

# 扩展知识

## 完整实现：并发队列

```javascript
class ConcurrentQueue {
  constructor(concurrency = 5) {
    this.concurrency = concurrency;
    this.queue = [];          // 等待队列
    this.active = 0;          // 正在执行的任务数
    this.results = [];        // 结果
    this.completed = 0;       // 已完成计数
    this.allSettled = false;  // 是否全部加入
    this.resolveAll = null;   // 全部完成的resolve
  }

  // 添加任务
  add(task) {
    return new Promise((resolve, reject) => {
      this.queue.push({ task, resolve, reject });
      this._process();
    });
  }

  // 批量添加任务
  async addAll(tasks) {
    const promises = tasks.map(task => this.add(task));
    return Promise.all(promises);
  }

  // 处理队列
  _process() {
    while (this.active < this.concurrency && this.queue.length > 0) {
      const { task, resolve, reject } = this.queue.shift();
      this.active++;

      Promise.resolve()
        .then(() => task())
        .then(resolve, reject)
        .finally(() => {
          this.active--;
          this.completed++;
          this._process();  // 处理下一个任务
        });
    }
  }

  // 获取当前状态
  get status() {
    return {
      active: this.active,
      pending: this.queue.length,
      completed: this.completed
    };
  }
}

// 使用示例
const queue = new ConcurrentQueue(3);
const tasks = Array.from({ length: 10 }, (_, i) => () =>
  new Promise(resolve => {
    setTimeout(() => {
      console.log(`任务 ${i} 完成`);
      resolve(i);
    }, Math.random() * 2000);
  })
);

queue.addAll(tasks).then(results => {
  console.log('所有任务完成:', results);
});
```

## 通用异步并发控制函数

```javascript
// 传入任务数组和并发限制，返回所有结果
async function asyncPool(iterable, limit, iteratorFn) {
  const tasks = [];
  const executing = new Set();

  for (const item of iterable) {
    // 创建任务
    const p = Promise.resolve().then(() => iteratorFn(item, tasks.length));
    tasks.push(p);
    executing.add(p);

    // 清理已完成的任务
    const clean = () => executing.delete(p);
    p.then(clean, clean);

    // 达到并发限制时等待
    if (executing.size >= limit) {
      await Promise.race(executing);
    }
  }

  return Promise.all(tasks);
}

// 使用
async function fetchUrls(urls, limit = 3) {
  return asyncPool(urls, limit, async (url) => {
    const response = await fetch(url);
    return response.json();
  });
}
```

## 基于Generator的并发控制

```javascript
async function concurrentGenerator(tasks, limit) {
  const iterator = tasks[Symbol.iterator]();
  const results = [];

  // 工作函数：不断从迭代器取任务执行
  async function worker() {
    for (const [index, task] of iterator) {
      // 实际上我们需要手动迭代
    }
    // 改进版本：
    while (true) {
      const { value, done } = iterator.next();
      if (done) break;
      const index = results.length;
      results[index] = Promise.resolve().then(() => value());
    }
  }

  // 启动limit个worker
  const workers = Array.from({ length: limit }, worker);
  await Promise.all(workers);
  return Promise.all(results);
}

// 更好的Generator版本
async function* asyncBatchProcess(items, limit, fn) {
  const results = [];
  const executing = [];

  for (const item of items) {
    const p = fn(item).then(result => {
      executing.splice(executing.indexOf(p), 1);
      return result;
    });
    executing.push(p);
    results.push(p);

    if (executing.length >= limit) {
      const done = await Promise.race(executing);
      yield done;
    }
  }

  // 等待剩余任务
  while (executing.length > 0) {
    yield await Promise.race(executing);
  }
}
```

## 带失败重试的并发控制

```javascript
class RobustConcurrentQueue {
  constructor(concurrency = 3, maxRetries = 3) {
    this.concurrency = concurrency;
    this.maxRetries = maxRetries;
    this.queue = [];
    this.active = 0;
    this.results = [];
  }

  add(task, retries = this.maxRetries) {
    return new Promise((resolve, reject) => {
      this.queue.push({ task, retries, resolve, reject });
      this._process();
    });
  }

  async _executeWithRetry({ task, retries, resolve, reject }) {
    this.active++;
    for (let attempt = 1; attempt <= retries; attempt++) {
      try {
        const result = await task();
        resolve(result);
        return;
      } catch (err) {
        console.warn(`任务失败，第${attempt}次尝试`);
        if (attempt === retries) {
          reject(err);
        }
        // 退避延迟
        await new Promise(r => setTimeout(r, Math.pow(2, attempt) * 100));
      }
    }
  }
}

// 使用
const queue = new RobustConcurrentQueue(3, 3);
queue.add(() => fetch('/api/unstable'));
```

```javascript
// 简化实现
function pLimit(concurrency) {
  const queue = [];
  let active = 0;

  async function next() {
    active++;
    const { fn, resolve, reject } = queue.shift();
    try {
      const result = await fn();
      resolve(result);
    } catch (err) {
      reject(err);
    } finally {
      active--;
      if (queue.length > 0) next();
    }
  }

  return function(fn) {
    return new Promise((resolve, reject) => {
      queue.push({ fn, resolve, reject });
      if (active < concurrency) next();
    });
  };
}
```

```javascript
async function uploadFiles(files, concurrency = 3) {
  const results = [];
  const queue = new ConcurrentQueue(concurrency);

  for (const file of files) {
    queue.add(async () => {
      const formData = new FormData();
      formData.append('file', file);
      const response = await fetch('/api/upload', {
        method: 'POST',
        body: formData
      });

// 2. 爬虫并发控制
class Spider {
  constructor(maxConcurrency = 5) {
    this.queue = new ConcurrentQueue(maxConcurrency);
    this.visited = new Set();
  }

  async crawl(url) {
    if (this.visited.has(url)) return;
    this.visited.add(url);

    const html = await this.queue.add(() => fetch(url).then(r => r.text()));
    const urls = extractLinks(html);

    // 递归爬取
    await Promise.all(
      urls
        .filter(u => !this.visited.has(u))
        .map(u => this.crawl(u))
    );
  }
}
```

## 各种方案的对比

| 方案 | 复杂度 | 灵活性 | 控制力度 | 推荐场景 |
|------|:-----:|:-----:|:--------:|---------|
| 简单Promise.race | 低 | 低 | 弱 | 快速实现 |
| ConcurrentQueue类 | 中 | 高 | 强 | 复杂场景 |
| asyncPool | 低 | 中 | 中 | 批量处理 |
| p-limit库 | 低 | 高 | 强 | 生产环境 |
